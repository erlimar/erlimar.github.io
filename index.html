Resumo profissional

Adoro programar! Por isso faço questão de dizer sou programador, não analista ou desenvolvedor, mas programador.

Acredito que criar softwares é uma arte, e olhar para uma sequência de códigos bem escrita (seja qual for a linguagem) é semelhante a deslumbrar-se com uma bela pintura.

É uma pena ver que o mercado atual não valoriza tanto esse profissional, e mais penoso ainda é constatar que os profissionais que estão se formando hoje (seja em faculdades ou no próprio mercado de trabalho) estão perdendo essa essência: que "programar é uma arte".

Mas acredito que posso contribuir para mudar essa história. Quero empreender em uma solução para esse problema, e descobrir que ainda é possível criar softwares com qualidade, seja em uma empresa que busca o mesmo objetivo, ou em um novo negócio com os parceiros certos.

Quer participar desse projeto? Então o que tem a dizer sobre este código abaixo:

namespace IoC
{
    public interface IService1
    {
        void Run();
        bool Running();
    }

    public interface IService2
    {
        void OtherMethod();
    }

    public interface IService3
    {
        void OtherMethod2();
    }

    public class MyService1 : IService1
    {
        public void Run(){}
        public bool Running(){
            return false;
        }
    }

    public class MyService2 : IService2
    {
        public void OtherMethod(){}
    }

    public class MyService3 : IService3
    {
        public void OtherMethod(){}
    }

    public class MyProccess
    {
        private IService1 service1_;
        private IService2 service2_;
        private IService3 service3_;

        private IService1 service1_2_;
        private IService2 service2_2_;
        private IService3 service3_2_;

        public MyProccess(IService1 service1)
        {
            this.service1_ = service1;
        }

        public IService2 Service2
        {
            get { return this.service2_; }
            set { this.service2_ = value; }
        }

        [InjectionTo(typeof(IService3))]
        public object Service3
        {
            get { return this.service3_; }
            set {
                if(value.GetType() != this.service3_.GetType()){
                    throw new InvalidCastException();
                }
                this.service3_ = (IService3)value;
            }
        }

        [InjectionMethod]
        public void IoC(IService1 service1, IService2 service2,
            IService3 service3)
        {
            this.service1_2_ = service1;
            this.service2_2_ = service2;
            this.service3_2_ = service3;
        }
    }

    class InjectionToAttribute : Attribute
    {
        public InjectionToAttribute(Type type, string name){}
    }

    class InjectionMethodAttribute : Attribute
    {
    }
}
